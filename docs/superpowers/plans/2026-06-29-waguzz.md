# Waguzz Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Waguzz — a self-hosted WhatsApp HTTP API service (like WAHA) in a single Docker container, with a React dashboard, JWT auth with refresh tokens, Baileys WA engine with auto-reconnect, and a static API key for mellyna-education server-to-server calls.

**Architecture:** Express backend (ESM + tsx runtime, no compile step) serves both the REST API and the compiled React frontend as static files. Baileys handles WA Web protocol with auth state persisted to a Docker volume so QR scan survives restarts. Two separate auth layers: JWT+refresh for the human dashboard, static API key for mellyna.

**Tech Stack:** Node.js 20, Express, `@whiskeysockets/baileys`, `jose` (JWT), React 18, Vite, React Router, axios, vitest + supertest, Docker multi-stage

**Spec:** `docs/superpowers/specs/2026-06-29-waguzz-design.md` (in mellyna-education repo)

```typescript
export type LogLevel = 'info' | 'error' | 'warn'

export interface LogEntry {
  ts: string
  level: LogLevel
  msg: string
}

const MAX = 500
const buffer: LogEntry[] = []

export function log(level: LogLevel, msg: string): void {
  const entry: LogEntry = { ts: new Date().toISOString(), level, msg }
  if (buffer.length >= MAX) buffer.shift()
  buffer.push(entry)
  const fn = level === 'error' ? console.error : level === 'warn' ? console.warn : console.log
  fn(`[${entry.ts}] [${level.toUpperCase()}] ${msg}`)
}

export function getLogs(): LogEntry[] {
  return [...buffer]
}

export function clearLogs(): void {
  buffer.splice(0, buffer.length)
}

export const logger = {
  info: (msg: string) => log('info', msg),
  warn: (msg: string) => log('warn', msg),
  error: (msg: string) => log('error', msg),
}
```

```typescript
import { Request, Response, NextFunction } from 'express'
import { verifyAccessToken } from './auth.js'

export async function requireDashboardAuth(req: Request, res: Response, next: NextFunction): Promise<void> {
  const header = req.headers.authorization
  if (!header?.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Missing token' })
    return
  }
  const valid = await verifyAccessToken(header.slice(7))
  if (!valid) {
    res.status(401).json({ error: 'Invalid or expired token' })
    return
  }
  next()
}

export function requireApiKey(req: Request, res: Response, next: NextFunction): void {
  const expected = process.env.WAGUZZ_API_KEY ?? ''
  if (!expected || req.headers['x-api-key'] !== expected) {
    res.status(401).json({ error: 'Invalid API key' })
    return
  }
  next()
}

export function requireApiKeyOrDashboard(req: Request, res: Response, next: NextFunction): void {
  const expected = process.env.WAGUZZ_API_KEY ?? ''
  if (expected && req.headers['x-api-key'] === expected) {
    next()
    return
  }
  requireDashboardAuth(req, res, next)
}
```

```typescript
import makeWASocket, { DisconnectReason, useMultiFileAuthState } from '@whiskeysockets/baileys'
import QRCode from 'qrcode'
import pino from 'pino'
import { existsSync, mkdirSync, rmSync } from 'fs'
import { join } from 'path'
import { logger } from './logger.js'

export type WAStatus = 'CONNECTED' | 'DISCONNECTED' | 'CONNECTING'

const AUTH_DIR = join(process.cwd(), 'data', 'auth_state')

let socket: ReturnType<typeof makeWASocket> | null = null
let status: WAStatus = 'DISCONNECTED'
let qrCode: string | null = null
let connectedPhone: string | null = null
let lastActivity: string | null = null
let reconnectTimer: ReturnType<typeof setTimeout> | null = null
let reconnectDelay = 2000
let stopped = false

function clearTimer(): void {
  if (reconnectTimer) {
    clearTimeout(reconnectTimer)
    reconnectTimer = null
  }
}

function scheduleReconnect(): void {
  if (stopped) return
  clearTimer()
  logger.info(`Reconnect scheduled in ${reconnectDelay / 1000}s`)
  reconnectTimer = setTimeout(() => {
    reconnectDelay = Math.min(reconnectDelay * 2, 60_000)
    connect()
  }, reconnectDelay)
}

export async function connect(): Promise<void> {
  if (socket) {
    try { (socket as any).end?.(undefined) } catch {}
    socket = null
  }

  status = 'CONNECTING'
  qrCode = null

  if (!existsSync(AUTH_DIR)) {
    mkdirSync(AUTH_DIR, { recursive: true })
  }

  const { state, saveCreds } = await useMultiFileAuthState(AUTH_DIR)

  socket = makeWASocket({
    auth: state,
    printQRInTerminal: false,
    logger: pino({ level: 'silent' }) as any,
  })

  socket.ev.on('creds.update', saveCreds)

  socket.ev.on('connection.update', async (update) => {
    const { connection, lastDisconnect, qr } = update

    if (qr) {
      qrCode = await QRCode.toDataURL(qr)
      status = 'DISCONNECTED'
      logger.info('QR code ready — scan with WhatsApp')
    }

    if (connection === 'open') {
      status = 'CONNECTED'
      qrCode = null
      reconnectDelay = 2000
      connectedPhone = (socket?.user?.id ?? '').split(':')[0] || null
      logger.info(`WA connected: ${connectedPhone}`)
    }

    if (connection === 'close') {
      const code = (lastDisconnect?.error as any)?.output?.statusCode as number | undefined
      logger.warn(`WA closed, code: ${code}`)

      if (code === DisconnectReason.loggedOut) {
        logger.error('WA logged out — deleting auth_state, QR required')
        status = 'DISCONNECTED'
        connectedPhone = null
        rmSync(AUTH_DIR, { recursive: true, force: true })
        // Do NOT reconnect — user must rescan QR
      } else if (!stopped) {
        status = 'CONNECTING'
        scheduleReconnect()
      }
    }
  })
}

export function disconnect(): void {
  stopped = true
  clearTimer()
  if (socket) {
    try { (socket as any).end?.(undefined) } catch {}
    socket = null
  }
  status = 'DISCONNECTED'
  qrCode = null
  connectedPhone = null
  logger.info('WA disconnected by user')
}

export function reconnect(): void {
  stopped = false
  reconnectDelay = 2000
  connect().catch(e => logger.error(`reconnect error: ${e}`))
}

export function getStatus(): { status: WAStatus; phone: string | null; lastActivity: string | null; qrAvailable: boolean } {
  return { status, phone: connectedPhone, lastActivity, qrAvailable: qrCode !== null }
}

export function getQR(): string | null {
  return qrCode
}

function normalizePhone(phone: string): string {
  return phone.replace(/\D/g, '').replace(/^0/, '62')
}

function randomDelay(minMs = 1000, maxMs = 3000): Promise<void> {
  return new Promise(r => setTimeout(r, Math.floor(Math.random() * (maxMs - minMs + 1)) + minMs))
}

export async function sendText(phone: string, message: string): Promise<boolean> {
  if (!socket || status !== 'CONNECTED') {
    logger.error(`sendText failed: not connected (status=${status})`)
    return false
  }
  try {
    await randomDelay()
    await socket.sendMessage(`${normalizePhone(phone)}@s.whatsapp.net`, { text: message })
    lastActivity = new Date().toISOString()
    logger.info(`Text sent to ${phone}`)
    return true
  } catch (e) {
    logger.error(`sendText error: ${e}`)
    return false
  }
}

export async function sendFile(
  phone: string,
  base64Data: string,
  filename: string,
  mimetype: string,
  caption: string,
): Promise<{ ok: true } | { ok: false; error: string }> {
  if (!socket || status !== 'CONNECTED') {
    return { ok: false, error: `WA not connected (status=${status})` }
  }
  try {
    await randomDelay()
    const jid = `${normalizePhone(phone)}@s.whatsapp.net`
    const buf = Buffer.from(base64Data, 'base64')

    if (mimetype.startsWith('image/')) {
      await socket.sendMessage(jid, { image: buf, caption, mimetype })
    } else if (mimetype.startsWith('video/')) {
      await socket.sendMessage(jid, { video: buf, caption, mimetype })
    } else if (mimetype.startsWith('audio/')) {
      await socket.sendMessage(jid, { audio: buf, mimetype, ptt: false })
    } else {
      await socket.sendMessage(jid, { document: buf, caption, mimetype, fileName: filename })
    }

    lastActivity = new Date().toISOString()
    logger.info(`File sent to ${phone}: ${filename}`)
    return { ok: true }
  } catch (e) {
    const msg = e instanceof Error ? e.message : String(e)
    logger.error(`sendFile error: ${msg}`)
    return { ok: false, error: msg }
  }
}
```

```typescript
import { Router, Request, Response } from 'express'
import { checkCredentials, signAccessToken, signRefreshToken, verifyRefreshToken } from '../auth.js'

const COOKIE = 'waguzz_refresh'
const IS_PROD = process.env.NODE_ENV === 'production'

export const authRouter = Router()

authRouter.post('/login', async (req: Request, res: Response) => {
  const { username, password } = req.body ?? {}
  if (!username || !password || !checkCredentials(String(username), String(password))) {
    res.status(401).json({ error: 'Invalid credentials' })
    return
  }
  const [accessToken, refreshToken] = await Promise.all([signAccessToken(), signRefreshToken()])
  res.cookie(COOKIE, refreshToken, {
    httpOnly: true,
    secure: IS_PROD,
    sameSite: 'strict',
    maxAge: parseInt(process.env.JWT_REFRESH_TTL ?? '604800') * 1000,
  })
  res.json({ accessToken })
})

  const token = req.cookies?.[COOKIE]

    res.status(401).json({ error: 'No refresh token' })
    return
  }
  if (!(await verifyRefreshToken(token))) {
    res.clearCookie(COOKIE)
    res.status(401).json({ error: 'Invalid refresh token' })
    return
  }
  res.json({ accessToken: await signAccessToken() })
})
```

```typescript
import axios from 'axios'
import { getToken, setToken, clearToken } from './auth-store'

const api = axios.create({ withCredentials: true })

api.interceptors.request.use(config => {
  const token = getToken()
  if (token) config.headers.Authorization = `Bearer ${token}`
  return config
})

// Queue of callers waiting for an in-progress refresh
let isRefreshing = false
let refreshQueue: Array<(token: string | null) => void> = []

api.interceptors.response.use(
  res => res,
  async error => {
    const original = error.config
    // Only retry on 401, only once per request
    if (error.response?.status !== 401 || original._retried) {
      return Promise.reject(error)
    }
    original._retried = true

    if (isRefreshing) {
      // Another request is already refreshing — queue and wait
      return new Promise((resolve, reject) => {
        refreshQueue.push(newToken => {
          if (newToken) {
            original.headers.Authorization = `Bearer ${newToken}`
            resolve(api(original))
          } else {
            reject(error)
          }
        })
      })
    }

    isRefreshing = true
    try {
      const res = await axios.post('/auth/refresh', {}, { withCredentials: true })
      const newToken: string = res.data.accessToken
      setToken(newToken)
      refreshQueue.forEach(cb => cb(newToken))
      refreshQueue = []
      original.headers.Authorization = `Bearer ${newToken}`
      return api(original)
    } catch {
      clearToken()
      refreshQueue.forEach(cb => cb(null))
      refreshQueue = []
      window.location.href = '/login'
      return Promise.reject(error)
    } finally {
      isRefreshing = false
    }
  },
)

export default api
```

```typescript
export async function sendWhatsApp(phone: string, message: string): Promise<boolean> {
  try {
    const { base } = cfg()
    const res = await fetch(`${base}/wa/send-text`, {
      method: 'POST',
      headers: authHeaders(),
      body: JSON.stringify({ phone: normalizePhone(phone), message }),
    })
    if (!res.ok) {
      const body = await res.text().catch(() => '')
      console.error(`[WAGUZZ] sendText failed ${res.status} for ${phone}: ${body}`)
      return false
    }
    return true
  } catch (e) {
    console.error('[WAGUZZ] sendText error:', e)
    return false
  }
}

export async function getSessionStatus(): Promise<string> {
  try {
    const { base, apiKey } = cfg()
    const res = await fetch(`${base}/wa/status`, {

      return 'UNKNOWN'
```
